#!/bin/bash

# Получаем текущий временной штамп
timestamp=$(date +"%Y-%m-%d_%H-%M-%S")

# Перебираем все переданные файлы
for file in "$@"; do
  # Проверяем, является ли файл текстовым файлом
  if [[ -f "$file" && "$file" == *.txt ]]; then
    # Добавляем временной штамп в конец файла
    echo "$timestamp" >> "$file"
    echo "Временной штамп добавлен в файл: $file"
  else
    echo "Ошибка: $file не является текстовым файлом"
  fi
done
